# 前言

## nginx.conf文件结构

```nginx
# 全局块，可以配置运行的用户、nginx进程数、全局日志配置
events {
  # 事件模块，可以设置单个进程最大连接数等信息
}
http{
  #http全局范围，可以设置监听端口、虚拟主机等
  server{
    # 路由模块
    location URL路径{
        
    }
  }
}
https{
  # 同http块
}
```

# 配置文件示例

```nginx
# 设置nginx线程数，一般根据CPU数量来，例如2个4核计为8
work_processes 8;
events{
  # 单个进程最大连接数
  worker_connections 1024;
}
http{
  # 配置负载均衡,权重为可选项，权值越高被分配到的几率越大
  upstream myserver1{
    server example1.com weight=5;
    server example2.com weight=3;
  }
  server {
    # 监听端口
    listen port;
    # 配置虚拟主机，支持域名和ip，多个域名之间用空格分隔
    server_name example.com example2.com 127.0.0.1;
    location /example {
      
    }
    
  }
}
```
